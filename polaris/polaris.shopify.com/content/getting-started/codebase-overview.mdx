## Technical Deep Dive

### Cross-Layer Communication

The framework excels at enabling communication between layers while maintaining independence:

```typescript
// Business Logic Layer (TypeScript)
import { Repository, EventBus } from '@cin7/typescript-sdk';

class ProductRepository extends Repository<Product> {
  async findAll(params?: QueryParams) {
    const products = await this.apiClient.get('/products', params);
    return this.mapResponse(products);
  }

  async create(data: CreateDTO<Product>) {
    const product = await this.apiClient.post('/products', data);
    EventBus.emit('product:created', product);
    return product;
  }
}

// UI Component Layer (ExtJS)
import { ExtDataGrid } from '@cin7/extjs-adapters';

const productGrid = ExtDataGrid.create({
  store: productStore,
  features: ['grouping', 'sorting', 'export'],
  columns: [
    { text: 'Name', dataIndex: 'name', flex: 1 },
    { text: 'Price', dataIndex: 'price', xtype: 'currencycolumn' },
    { text: 'Status', dataIndex: 'status', xtype: 'statuscolumn' }
  ]
});

// UI Interaction Layer (Vanilla JS)
import { on, addClass, fadeOut } from '@cin7/vanilla-js';

on('#add-product', 'click', async () => {
  addClass('#loading', 'visible');
  try {
    const formData = getFormData();
    const product = await productRepo.create(formData);
    showToast('Product added successfully');
  } catch (error) {
    showToast('Error adding product', 'error');
  } finally {
    fadeOut('#loading');
  }
});

// Cross-layer Communication
EventBus.on('product:created', (product) => {
  productGrid.getStore().add(product);
  updateProductCount();
});
```
