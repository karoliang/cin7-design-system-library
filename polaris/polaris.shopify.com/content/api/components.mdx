## Core Components

### Button

A button triggers an action when clicked.

```typescript
interface ButtonProps {
  /** The content to display inside the button */
  children?: React.ReactNode;
  /** Makes button primary (filled) */
  primary?: boolean;
  /** Changes the tone of the button */
  tone?: 'critical' | 'success';
  /** Makes the button a plain text button */
  plain?: boolean;
  /** Shows a loading spinner */
  loading?: boolean;
  /** Disables the button */
  disabled?: boolean;
  /** Makes button take full width */
  fullWidth?: boolean;
  /** Button size */
  size?: 'slim' | 'medium' | 'large';
  /** Icon to display */
  icon?: React.ComponentType<{className?: string}>;
  /** Callback when clicked */
  onClick?: () => void;
  /** URL to link to */
  url?: string;
  /** Opens URL in new tab */
  external?: boolean;
  /** Accessibility label */
  ariaLabel?: string;
}
```

#### Examples

```tsx
// Primary button
<Button primary onClick={handleSave}>Save</Button>

// Critical button
<Button tone="critical" onClick={handleDelete}>Delete</Button>

// Loading state
<Button loading>Processing...</Button>

// With icon
<Button icon={PlusIcon}>Add Product</Button>
```

### Card

Cards are used to group similar concepts and tasks together.

```typescript
interface CardProps {
  /** Title content for the card */
  title?: React.ReactNode;
  /** Card sections */
  children?: React.ReactNode;
  /** Actions in the card header */
  actions?: ActionListItemDescriptor[];
  /** Footer actions */
  footerActions?: ComplexAction[];
  /** Remove outer padding */
  flush?: boolean;
  /** Subdued background color */
  subdued?: boolean;
  /** Section dividers */
  sectioned?: boolean;
  /** Round the corners */
  roundedAbove?: 'sm' | 'md' | 'lg' | 'xl';
}
```

#### Examples

```tsx
// Basic card
<Card title="Customer details">
  <p>John Doe</p>
</Card>

// Card with actions
<Card
  title="Product"
  actions={[{content: 'Edit', onAction: handleEdit}]}
  footerActions={[{content: 'Delete', destructive: true}]}
>
  <p>Product information</p>
</Card>
```

### TextField

Text fields allow users to input text.

```typescript
interface TextFieldProps {
  /** Label for the input */
  label?: string;

  /** Change handler */
  onChange?: (value: string) => void;

  type?: 'text' | 'email' | 'number' | 'password' | 'search' | 'tel' | 'url';

  /** Disable input */
  disabled?: boolean;
  /** Required field */
  requiredIndicator?: boolean;
  /** Character count */
  showCharacterCount?: boolean;
  /** Max length */
  maxLength?: number;
  /** Auto complete */
  autoComplete?: string;
  /** Multi-line */
  multiline?: boolean | number;
  /** Prefix */
  prefix?: React.ReactNode;
  /** Suffix */
  suffix?: React.ReactNode;
}
```

#### Examples

```tsx
// Basic text field
<TextField
  label="Email"
  type="email"
  value={email}
  onChange={setEmail}
  autoComplete="email"
/>

// With validation
<TextField
  label="Password"
  type="password"
  value={password}
  onChange={setPassword}
  error={password.length < 8 ? "Password must be at least 8 characters" : undefined}
  helpText="Use 8 or more characters"
/>

// With prefix/suffix
<TextField
  label="Price"
  type="number"
  prefix="$"
  suffix="USD"
  value={price}
  onChange={setPrice}
/>
```

## Hooks

### useDebounce

Debounce a value or function.

```typescript
function useDebounce<T>(value: T, delay: number): T;
```

#### Example

```tsx
function SearchComponent() {
  const [searchTerm, setSearchTerm] = useState('');
  const debouncedSearchTerm = useDebounce(searchTerm, 500);

  useEffect(() => {
    if (debouncedSearchTerm) {
      // Perform search
      performSearch(debouncedSearchTerm);
    }
  }, [debouncedSearchTerm]);

  return (
    <TextField
      label="Search"
      value={searchTerm}
      onChange={setSearchTerm}
    />
  );
}
```

### useMediaQuery

Respond to media query changes.

```typescript
function useMediaQuery(query: string): boolean;
```

#### Example

```tsx
function ResponsiveComponent() {
  const isMobile = useMediaQuery('(max-width: 768px)');
  const isTablet = useMediaQuery('(min-width: 769px) and (max-width: 1024px)');

  return (
    <Layout>
      {isMobile ? (
        <MobileView />
      ) : isTablet ? (
        <TabletView />
      ) : (
        <DesktopView />
      )}
    </Layout>
  );
}
```

### useAsync

Handle async operations with loading and error states.

```typescript
interface UseAsyncReturn<T> {
  data: T | null;
  loading: boolean;
  error: Error | null;
  execute: (...args: any[]) => Promise<T>;
  reset: () => void;
}

function useAsync<T>(
  asyncFunction: (...args: any[]) => Promise<T>
): UseAsyncReturn<T>;
```

#### Example

```tsx
function UserProfile({ userId }) {
  const {
    data: user,
    loading,
    error,
    execute: loadUser
  } = useAsync(() => fetchUser(userId));

  useEffect(() => {
    loadUser();
  }, [userId]);

  if (loading) return <Spinner />;
  if (error) return <Banner tone="critical">{error.message}</Banner>;
  if (!user) return null;

  return <UserCard user={user} />;
}
```

## Utilities

### classNames

Conditionally join class names.

```typescript
function classNames(
  ...args: (string | undefined | null | false | {[key: string]: boolean})[]
): string;
```

#### Example

```tsx
const buttonClass = classNames(
  'btn',
  primary && 'btn-primary',
  disabled && 'btn-disabled',
  {
    'btn-large': size === 'large',
    'btn-small': size === 'small',
  }
);
```

### formatCurrency

Format numbers as currency.

```typescript
function formatCurrency(
  amount: number,
  currency?: string,
  locale?: string
): string;
```

#### Example

```tsx
formatCurrency(1234.56); // "$1,234.56"
formatCurrency(1234.56, 'EUR'); // "€1,234.56"
formatCurrency(1234.56, 'JPY', 'ja-JP'); // "¥1,235"
```
